from dotenv import load_dotenv
from google import genai
import os

load_dotenv()

client = genai.Client(api_key=os.getenv("GEMINI_API_KEY"))

def main():
    fileName = input("Enter File To Interpret: ")
    try:
        with open(fileName, "r", encoding="utf-8") as f:
            data = f.read()
    except Exception as e:
        print(f"Error reading file: {e}")
        return

    instruction = input("Enter Command for Gemini: ")

    try:
        response = client.models.generate_content(
            model="gemini-2.0-flash", # Updated model name
            contents=[instruction, data],
            config={
                "system_instruction": "You are a professional data analyst. Be concise.",
                "temperature": 0.3
            }
        )

        with open("output.txt", "w", encoding="utf-8") as fOut:
            fOut.write(response.text)
        print("Analysis saved to output.txt")
        
    except Exception as e:
        print(f"API Error: {e}")

if __name__ == "__main__":
    main()
